# useSignatureStatuses

The `useSignatureStatuses` hook checks the confirmation status of multiple transaction signatures simultaneously. This is essential for tracking transaction confirmations, handling failures, and providing real-time status updates to users.

## Usage

### Direct Import
```tsx
import { useSignatureStatuses } from '@connectorkit/solana'

function TransactionStatusTracker() {
  const signatures = ['signature1...', 'signature2...'] // Your transaction signatures
  
  const { statuses, isLoading, error } = useSignatureStatuses({
    signatures,
    refreshInterval: 5000
  })
  
  if (isLoading) return <div>Checking transaction status...</div>
  if (error) return <div>Error: {error.message}</div>
  
  return (
    <div>
      {statuses?.map((status, index) => (
        <div key={index}>
          Signature {index + 1}: {status?.confirmationStatus || 'Pending'}
        </div>
      ))}
    </div>
  )
}
```

### Copy Hook Code
```tsx
'use client'

import { useQuery, useQueryClient } from '@tanstack/react-query'
import { useCallback } from 'react'
import { useArcClient } from '../core/arc-client-provider'
import type { Transport } from '../transports/types'
import { 
  ArcError,
  createNetworkError,
  type ArcErrorContext
} from '../core/error-handler'
import type { Signature } from '@solana/keys'

type SignatureStatus = Readonly<{
    slot: bigint;
    confirmations: bigint | null;
    confirmationStatus: 'processed' | 'confirmed' | 'finalized' | null;
    err: unknown | null;
}> | null;

export interface UseSignatureStatusesOptions {
  signatures: Signature[]
  refreshInterval?: number
  enabled?: boolean
}

export interface UseSignatureStatusesReturn {
  statuses: readonly SignatureStatus[] | null
  isLoading: boolean
  error: ArcError | null
  refetch: () => void
  clear: () => void
}

export function useSignatureStatuses(
  options: UseSignatureStatusesOptions
): UseSignatureStatusesReturn {
  const { network } = useArcClient()
  const queryClient = useQueryClient()
  
  const {
    signatures,
    refreshInterval,
    enabled = true,
  } = options

  const query = useQuery<readonly SignatureStatus[], ArcError>({
    networkMode: "offlineFirst",
    queryKey: ['signatureStatuses', network.rpcUrl, signatures],
    queryFn: async ({ signal }) => {
      const context: ArcErrorContext = {
        operation: 'getSignatureStatuses',
        timestamp: Date.now(),
        rpcUrl: network.rpcUrl
      }

      try {
        const transport = (useArcClient() as any).config.transport as Transport
        const response: any = await transport.request(
          { method: 'getSignatureStatuses', params: [signatures] },
          { signal }
        )
        return response?.value
      } catch (error) {
        const errorMessage = error instanceof Error ? error.message : String(error)
        throw createNetworkError(
          `Failed to fetch signature statuses: ${errorMessage}`,
          context,
          error as Error
        )
      }
    },
    enabled: enabled && signatures.length > 0,
    refetchInterval: refreshInterval,
    refetchOnWindowFocus: false,
    staleTime: 5_000,
    gcTime: 5 * 60_000,
    retry: 1,
    notifyOnChangeProps: ['data', 'error', 'isLoading'],
  })

  const clear = useCallback(() => {
    const key = ['signatureStatuses', network.rpcUrl, signatures] as const
    queryClient.cancelQueries({ queryKey: key, exact: true }).finally(() => {
      queryClient.removeQueries({ queryKey: key, exact: true })
    })
  }, [network.rpcUrl, signatures, queryClient])

  return {
    statuses: query.data ?? null,
    isLoading: query.isLoading,
    error: query.error ?? null,
    refetch: query.refetch,
    clear,
  }
}
```

## Parameters

- `signatures: Signature[]` - Array of transaction signatures to check (required)
- `refreshInterval?: number` - How often to refresh data in milliseconds (optional)
- `enabled?: boolean` - Whether the query is enabled (default: true)

## Return Value

- `statuses: readonly SignatureStatus[] | null` - Array of signature statuses (same order as input signatures):
  - `slot: bigint` - Slot where the transaction was processed
  - `confirmations: bigint | null` - Number of confirmations (null if not found)
  - `confirmationStatus: 'processed' | 'confirmed' | 'finalized' | null` - Current confirmation level
  - `err: unknown | null` - Error if transaction failed, null if successful
- `isLoading: boolean` - Whether the statuses are loading
- `error: ArcError | null` - Any error that occurred
- `refetch: () => void` - Function to manually refetch the statuses
- `clear: () => void` - Function to clear cached data and stop tracking

## Example

```tsx
import { useSignatureStatuses } from '@connectorkit/solana'
import { useState, useEffect } from 'react'

function TransactionTracker() {
  const [signatures, setSignatures] = useState<string[]>([
    // Add your transaction signatures here
    '3hZ8JQk4LxNJrJKJ4SKJS...', // Example signatures
    '2gY7HFK3MpOKJ5JSLKJS...',
  ])

  const {
    statuses,
    isLoading,
    error,
    refetch,
    clear
  } = useSignatureStatuses({
    signatures,
    refreshInterval: 3000 // Check every 3 seconds
  })

  const [notifications, setNotifications] = useState<string[]>([])

  useEffect(() => {
    if (!statuses) return

    statuses.forEach((status, index) => {
      const signature = signatures[index]
      if (!signature) return

      if (status?.confirmationStatus === 'finalized') {
        const message = status.err 
          ? `Transaction ${signature.slice(0, 8)}... failed`
          : `Transaction ${signature.slice(0, 8)}... finalized successfully`
        
        setNotifications(prev => {
          if (!prev.includes(message)) {
            return [...prev, message]
          }
          return prev
        })
      }
    })
  }, [statuses, signatures])

  const addSignature = () => {
    const newSig = prompt('Enter transaction signature:')
    if (newSig && !signatures.includes(newSig)) {
      setSignatures(prev => [...prev, newSig])
    }
  }

  const removeSignature = (index: number) => {
    setSignatures(prev => prev.filter((_, i) => i !== index))
  }

  const getStatusColor = (status: any) => {
    if (!status) return '#999'
    if (status.err) return '#f44336'
    
    switch (status.confirmationStatus) {
      case 'processed': return '#ff9800'
      case 'confirmed': return '#2196f3'
      case 'finalized': return '#4caf50'
      default: return '#999'
    }
  }

  const getStatusText = (status: any) => {
    if (!status) return 'Not Found'
    if (status.err) return 'Failed'
    
    switch (status.confirmationStatus) {
      case 'processed': return 'Processed'
      case 'confirmed': return 'Confirmed'
      case 'finalized': return 'Finalized'
      default: return 'Unknown'
    }
  }

  if (isLoading && !statuses) {
    return (
      <div style={{ padding: '16px', textAlign: 'center' }}>
        <div>Loading transaction statuses...</div>
      </div>
    )
  }

  if (error) {
    return (
      <div style={{ padding: '16px', color: 'red' }}>
        <div>Error loading statuses: {error.message}</div>
        <button onClick={() => refetch()}>Retry</button>
      </div>
    )
  }

  return (
    <div style={{ padding: '16px' }}>
      <h3>Transaction Status Tracker</h3>
      
      {notifications.length > 0 && (
        <div style={{ 
          marginBottom: '16px',
          padding: '12px',
          backgroundColor: '#f0f4ff',
          borderRadius: '8px',
          border: '1px solid #2196f3'
        }}>
          <h4 style={{ margin: '0 0 8px 0' }}>Recent Updates</h4>
          {notifications.slice(-3).map((notification, index) => (
            <div key={index} style={{ fontSize: '14px', color: '#333' }}>
              {notification}
            </div>
          ))}
        </div>
      )}

      <div style={{ display: 'grid', gap: '12px', marginBottom: '16px' }}>
        {signatures.map((signature, index) => {
          const status = statuses?.[index]
          const statusColor = getStatusColor(status)
          const statusText = getStatusText(status)
          
          return (
            <div key={index} style={{ 
              backgroundColor: '#f5f5f5', 
              padding: '16px', 
              borderRadius: '8px',
              border: `2px solid ${statusColor}`
            }}>
              <div style={{ display: 'flex', justifyContent: 'space-between', alignItems: 'start' }}>
                <div style={{ flex: 1 }}>
                  <div style={{ 
                    fontFamily: 'monospace', 
                    fontSize: '12px',
                    wordBreak: 'break-all',
                    marginBottom: '8px'
                  }}>
                    <strong>Signature:</strong> {signature}
                  </div>
                  
                  <div style={{ display: 'grid', gap: '4px', fontSize: '14px' }}>
                    <div>
                      <strong>Status:</strong> 
                      <span style={{ color: statusColor, marginLeft: '8px' }}>
                        {statusText}
                      </span>
                    </div>
                    
                    {status && (
                      <>
                        <div>
                          <strong>Slot:</strong> {status.slot.toString()}
                        </div>
                        
                        {status.confirmations !== null && (
                          <div>
                            <strong>Confirmations:</strong> {status.confirmations.toString()}
                          </div>
                        )}
                        
                        {status.err && (
                          <div style={{ color: '#f44336' }}>
                            <strong>Error:</strong> {JSON.stringify(status.err)}
                          </div>
                        )}
                      </>
                    )}
                  </div>
                </div>
                
                <button 
                  onClick={() => removeSignature(index)}
                  style={{
                    padding: '4px 8px',
                    backgroundColor: '#f44336',
                    color: 'white',
                    border: 'none',
                    borderRadius: '4px',
                    cursor: 'pointer',
                    fontSize: '12px'
                  }}
                >
                  Remove
                </button>
              </div>
            </div>
          )
        })}
        
        {signatures.length === 0 && (
          <div style={{ 
            padding: '32px',
            textAlign: 'center',
            backgroundColor: '#f5f5f5',
            borderRadius: '8px',
            color: '#666'
          }}>
            No transaction signatures being tracked.
            <br />
            Click "Add Signature" to start tracking transactions.
          </div>
        )}
      </div>

      <div style={{ display: 'flex', gap: '8px', flexWrap: 'wrap' }}>
        <button 
          onClick={addSignature}
          style={{
            padding: '8px 16px',
            backgroundColor: '#4CAF50',
            color: 'white',
            border: 'none',
            borderRadius: '4px',
            cursor: 'pointer'
          }}
        >
          Add Signature
        </button>
        
        <button 
          onClick={() => refetch()}
          style={{
            padding: '8px 16px',
            backgroundColor: '#2196F3',
            color: 'white',
            border: 'none',
            borderRadius: '4px',
            cursor: 'pointer'
          }}
        >
          Refresh All
        </button>
        
        <button 
          onClick={() => clear()}
          style={{
            padding: '8px 16px',
            backgroundColor: '#ff9800',
            color: 'white',
            border: 'none',
            borderRadius: '4px',
            cursor: 'pointer'
          }}
        >
          Clear & Stop
        </button>
        
        <button 
          onClick={() => setNotifications([])}
          style={{
            padding: '8px 16px',
            backgroundColor: '#9E9E9E',
            color: 'white',
            border: 'none',
            borderRadius: '4px',
            cursor: 'pointer'
          }}
        >
          Clear Notifications
        </button>
      </div>
    </div>
  )
}
```

## Confirmation Levels

**Processed:**
- Transaction has been included in a block
- Lowest confirmation level
- Can still be rolled back in case of forks

**Confirmed:**
- Transaction has been confirmed by supermajority of validators
- More secure than processed
- Very unlikely to be rolled back

**Finalized:**
- Transaction is permanently confirmed
- Cannot be rolled back
- Highest security level

## Transaction Status Flow

1. **Submitted** → Transaction sent to network
2. **Processed** → Included in a block (1-2 seconds)
3. **Confirmed** → Supermajority agreement (~13 seconds)
4. **Finalized** → Permanent confirmation (~32 seconds)

## Use Cases

- **Transaction Monitoring**: Track multiple transactions simultaneously
- **Wallet Applications**: Show real-time confirmation progress
- **DeFi Protocols**: Ensure transaction completion before proceeding
- **Payment Systems**: Verify payment finality
- **Trading Applications**: Confirm order execution status

## Best Practices

- **Batch Checking**: Check multiple signatures together for efficiency
- **Appropriate Polling**: Use reasonable refresh intervals (3-5 seconds)
- **Handle Errors**: Check for transaction errors in the `err` field
- **Confirmation Requirements**: Choose appropriate confirmation level for your use case
- **Cleanup**: Use `clear()` to stop tracking completed transactions

## Notes

- Returns statuses in the same order as input signatures
- `null` status means transaction was not found (may still be pending)
- Use shorter refresh intervals for time-sensitive applications
- Failed transactions will have an `err` field with error details
- Finalized status provides the highest confidence of permanence
- Consider using WebSocket connections for real-time updates in production
